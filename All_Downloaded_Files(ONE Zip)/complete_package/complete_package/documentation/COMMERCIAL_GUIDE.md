```bash

wget https://releases.najafcemetery.com/latest/installer.tar.gz
tar xzf installer.tar.gz
cd najaf-cemetery-installer

# 2. Configure
cp config.example.yml config.yml
nano config.yml  # Add license key

# 3. Run installer
sudo ansible-playbook deploy.yml --extra-vars "license_key=YOUR_KEY"

# 4. Verify
```
